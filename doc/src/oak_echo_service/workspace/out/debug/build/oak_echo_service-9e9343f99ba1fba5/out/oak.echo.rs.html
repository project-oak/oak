#[derive(Clone, PartialEq, ::prost::Message)]
pub struct EchoRequest {
    #[prost(bytes = "vec", tag = "1")]
    pub body: ::prost::alloc::vec::Vec<u8>,
}
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct EchoResponse {
    #[prost(bytes = "vec", tag = "1")]
    pub body: ::prost::alloc::vec::Vec<u8>,
}
#[derive(Clone)]
pub struct EchoServer<S> {
    service: S,
}
impl<S: Echo> ::micro_rpc::Transport for EchoServer<S> {
    fn invoke(
        &mut self,
        request_bytes: &[u8],
    ) -> Result<::prost::alloc::vec::Vec<u8>, !> {
        let response_bytes = self
            .invoke_inner(request_bytes)
            .map_or_else(::micro_rpc::error_response, ::micro_rpc::success_response)
            .encode_to_vec();
        Ok(response_bytes)
    }
}
impl<S: Echo> EchoServer<S> {
    pub fn new(service: S) -> Self {
        Self { service }
    }
    fn invoke_inner(
        &mut self,
        request_bytes: &[u8],
    ) -> Result<::prost::alloc::vec::Vec<u8>, ::micro_rpc::Status> {
        let request = ::micro_rpc::Request::decode(request_bytes)
            .map_err(|err| {
                ::micro_rpc::Status::new_with_message(
                    ::micro_rpc::StatusCode::Internal,
                    ::micro_rpc::format!(
                        "Client failed to deserialize the response: {:?}", err
                    ),
                )
            })?;
        match request.method_id {
            1988 => {
                let request = <EchoRequest>::decode(request.body.as_ref())
                    .map_err(|err| {
                        ::micro_rpc::Status::new_with_message(
                            ::micro_rpc::StatusCode::Internal,
                            ::micro_rpc::format!(
                                "Service failed to deserialize the request: {:?}", err
                            ),
                        )
                    })?;
                let response = self.service.echo(&request)?;
                let response_body = response.encode_to_vec();
                Ok(response_body)
            }
            _ => Err(::micro_rpc::Status::new(::micro_rpc::StatusCode::Unimplemented)),
        }
    }
}
pub trait Echo: Sized {
    fn echo(
        &mut self,
        request: &EchoRequest,
    ) -> Result<EchoResponse, ::micro_rpc::Status>;
}
pub struct EchoClient<T: ::micro_rpc::Transport> {
    transport: T,
}
impl<T: ::micro_rpc::Transport> EchoClient<T> {
    pub fn new(transport: T) -> Self {
        Self { transport }
    }
    pub fn echo(
        &mut self,
        request: &EchoRequest,
    ) -> Result<Result<EchoResponse, ::micro_rpc::Status>, T::Error> {
        ::micro_rpc::client_invoke(&mut self.transport, 1988, request)
    }
}
pub struct EchoAsyncClient<T: ::micro_rpc::AsyncTransport> {
    transport: T,
}
impl<T: ::micro_rpc::AsyncTransport> EchoAsyncClient<T> {
    pub fn new(transport: T) -> Self {
        Self { transport }
    }
    pub async fn echo(
        &mut self,
        request: &EchoRequest,
    ) -> Result<Result<EchoResponse, ::micro_rpc::Status>, T::Error> {
        ::micro_rpc::async_client_invoke(&mut self.transport, 1988, request).await
    }
}
