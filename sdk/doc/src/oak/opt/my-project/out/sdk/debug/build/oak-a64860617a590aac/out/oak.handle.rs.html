/// Sender handle for an Oak channel.
/// This type is sent over the wire, but in types generated by prost it is replaced with
/// [`oak::io::Sender`](../io/struct.Sender.html).
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct Sender {
    #[prost(fixed64, tag="1")]
    pub id: u64,
}
/// Receiver handle for an Oak channel.
/// This type is sent over the wire, but in types generated by prost it is replaced with
/// [`oak::io::Receiver`](../io/struct.Receiver.html).
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct Receiver {
    #[prost(fixed64, tag="1")]
    pub id: u64,
}
